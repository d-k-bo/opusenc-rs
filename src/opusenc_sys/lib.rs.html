13/// Read an encoder option.
14///
15/// libopus/libopusenc uses C macros to get options.
16///
17/// A `$PREFIX_GET_$X(x)` macro usually resolves to `$PREFIX_GET_$X_REQUEST, x` where x is checked to be a valid pointer
18#[inline]
19pub unsafe fn ope_encoder_get_option(enc: *mut OggOpusEnc, request: u32, x: *mut c_int) -> c_int {
20    ope_encoder_ctl(enc, request as c_int, x)
21}
