4//! High-level bindings for
5//! [libopusenc](https://opus-codec.org/docs/libopusenc_api-0.2/index.html).
6//!
7//! # Example
8//!
9//! ```
10//! # use std::io::Read;
11//! # use opusenc::{Comments, Encoder, MappingFamily, RecommendedTag};
12//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
13//! let audio_data: Vec<i16> = {
14//!     let mut file = std::fs::File::open("/dev/urandom")?;
15//!     let mut buf = vec![0; 60 * 48_000 * 2 * 2];
16//!     file.read_exact(&mut buf)?;
17//!     buf.chunks_exact(2)
18//!         .map(|a| i16::from_ne_bytes([a[0], a[1]]))
19//!         .collect()
20//! };
21//!
22//! let mut encoder = Encoder::create_file(
23//!     "/tmp/noise.opus",
24//!     Comments::create()
25//!         .add(RecommendedTag::Title, "Random Noise")?
26//!         .add(RecommendedTag::Artist, "/dev/urandom")?,
27//!     48_000,
28//!     2,
29//!     MappingFamily::MonoStereo,
30//! )?;
31//!     
32//! encoder.write(&audio_data)?;
33//! encoder.drain()?;
34//!
35//! # Ok(())
36//! # }
37//! ```
38//!
39//! # Encoder options
40//!
41//! This crate provides a `encoder-options` feature which enables reading and changing encoder options.
42//!
43//! **Warning:** Some of these options might not work with opusenc, may be unsafe or even cause UB.
44//! They are intended to be used via C macros that don't work with Rust.
45//! Make sure to check if the methods you use match their intended behaviour.
46

62/// Returns a string representing the version of libopusenc being used at run time.
63pub fn version_str() -> &'static CStr {
64    unsafe { CStr::from_ptr(crate::ffi::ope_get_version_string()) }
65}
66
67/// ABI version for libopusenc's header.
68/// Can be used to check for features at run time.
69pub fn abi_version() -> i32 {
70    unsafe { crate::ffi::ope_get_abi_version() }
