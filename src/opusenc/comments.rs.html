13/// Comments to be attached to an Ogg Opus file.
14///
15/// Can contain metadata in [VorbisComment](https://wiki.xiph.org/VorbisComment)
16/// format and images.
17///
18/// # Example
19///
20/// ```
21/// use opusenc::{Comments, PicardTag, RecommendedTag};
22///
23/// # fn main() -> opusenc::Result<()> {
24/// let comments = Comments::create()
25///     .add(RecommendedTag::Title, "The Hardest Button to Button")?
26///     .add(RecommendedTag::Artist, "The White Stripes")?
27///     .add(RecommendedTag::Album, "Elephant")?
28///     .add(RecommendedTag::TrackNumber, "9")?
29///     .add(PicardTag::AlbumArtist, "The White Stripes")?
30///     .add(RecommendedTag::Isrc, "USVT10300009")?;
31/// # Ok(())
32/// # }
33/// ```
34pub struct Comments {
35    ptr: *mut crate::ffi::OggOpusComments,
36}
37
38impl Drop for Comments {
39    fn drop(&mut self) {
40        unsafe { crate::ffi::ope_comments_destroy(self.ptr) };
41    }
42}
43
44impl Comments {
45    /// Create a new comments object.
46    pub fn create() -> Self {
47        let ptr = unsafe { crate::ffi::ope_comments_create() };
48        assert!(!ptr.is_null());
49        Self { ptr }
50    }
51}
52
53impl Comments {
54    /// Add a comment.
55    ///
56    /// For common tags, see [`RecommendedTag`] and [`PicardTag`].
57    ///
58    /// Tag names must not contain the `=` charater.
59    pub fn add(&mut self, tag: impl AsRef<CStr>, val: impl Into<Vec<u8>>) -> Result<&mut Self> {
60        let val = CString::new(val)?;
61        unsafe { crate::ffi::ope_comments_add(self.ptr, tag.as_ref().as_ptr(), val.as_ptr()) }
62            .check_result()?;
63
64        Ok(self)
65    }
66    /// Add a comment as a single tag=value string.
67    ///
68    /// The string must contain the `=` character.
69    pub fn add_string(&mut self, tag_and_val: impl Into<Vec<u8>>) -> Result<&mut Self> {
70        let tag_and_val = CString::new(tag_and_val)?;
71        unsafe { crate::ffi::ope_comments_add_string(self.ptr, tag_and_val.as_ptr()) }
72            .check_result()?;
73
74        Ok(self)
75    }
76    /// Add a picture from a file.
77    pub fn add_picture(
78        &mut self,
79        filename: impl AsRef<Path>,
80        picture_type: PictureType,
81        description: Option<impl Into<Vec<u8>>>,
82    ) -> Result<&mut Self> {
83        let filename = CString::new(filename.as_ref().as_os_str().as_bytes())?;
84        let description = description.map(CString::new).transpose()?;
85        unsafe {
86            crate::ffi::ope_comments_add_picture(
87                self.ptr,
88                filename.as_ptr(),
89                picture_type as i32,
90                description.map(|d| d.as_ptr()).unwrap_or(std::ptr::null()),
91            )
92        }
93        .check_result()?;
94
95        Ok(self)
96    }
97    /// Add a picture already in memory.
98    pub fn add_picture_from_memory(
99        &mut self,
100        picture: impl AsRef<[u8]>,
101        picture_type: PictureType,
102        description: Option<impl Into<Vec<u8>>>,
103    ) -> Result<&mut Self> {
104        let picture = picture.as_ref();
105        let description = description.map(CString::new).transpose()?;
106        unsafe {
107            crate::ffi::ope_comments_add_picture_from_memory(
108                self.ptr,
109                picture.as_ptr() as *const i8,
110                picture.len(),
111                picture_type as i32,
112                description.map(|d| d.as_ptr()).unwrap_or(std::ptr::null()),
113            )
114        }
115        .check_result()?;
116
117        Ok(self)
118    }
119}
120
121impl Comments {
122    pub fn ptr(&self) -> *mut crate::ffi::OggOpusComments {
123        self.ptr
124    }
125}

127impl Clone for Comments {
128    fn clone(&self) -> Self {
129        let ptr = unsafe { crate::ffi::ope_comments_copy(self.ptr) };
130        assert!(!ptr.is_null());
131        Self { ptr }
132    }
133}

135impl Default for Comments {
136    fn default() -> Self {
137        Comments::create()
138    }
139}

190macro_rules! tag_names {
191    {
192        $( #[$enum_attr:meta] )*
193        $vis:vis enum $enum_name:ident {
194            $( $(#[$attr:meta] )*
195            $variant:ident = $val:literal ),* $(,)?
196        }
197    } => {
198        $(#[$enum_attr])*
199        $vis enum $enum_name {
200            $(
201                $( #[$attr] )*
202                $variant,
203            )*
204        }
205        impl $enum_name {
206            const fn as_cstr8(&self) -> &'static cstr8::CStr8 {
207                match self {
208                    $(
209                        $enum_name::$variant => cstr8::cstr8!($val),
210                    )*
211                }
212            }
213            $vis const fn as_str(&self) -> &'static str {
214                self.as_cstr8().as_str()
215            }
216            $vis const fn as_cstr(&self) -> &'static CStr {
217                self.as_cstr8().as_c_str()
218            }
219        }
220        impl AsRef<str> for $enum_name {
221            #[inline]
222            fn as_ref(&self) -> &str {
223                self.as_str()
224            }
225        }
226        impl AsRef<CStr> for $enum_name {
227            #[inline]
228            fn as_ref(&self) -> &CStr {
229                self.as_cstr()
230            }
231        }
232        impl Display for $enum_name {
233            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
234                f.write_str(self.as_str())
235            }
236        }
237    };
238}

240tag_names! {
241    /// Tags recommended by the Ogg Vorbis specification.
242    ///
243    /// <https://xiph.org/vorbis/doc/v-comment.html>
244    #[derive(Copy, Clone, Debug)]
245    #[non_exhaustive]
246    pub enum RecommendedTag {
247        /// Track/Work name
248        Title = "TITLE",
249        /// The version field may be used to differentiate multiple versions of the same track title in a single collection. (e.g. remix info)
250        Version = "VERSION",
251        /// The collection name to which this track belongs
252        Album = "ALBUM",
253        /// The track number of this piece if part of a specific larger collection or album
254        TrackNumber = "TRACKNUMBER",
255        /// The artist generally considered responsible for the work. In popular music this is usually the performing band or singer. For classical music it would be the composer. For an audio book it would be the author of the original text.
256        Artist = "ARTIST",
257        /// The artist(s) who performed the work. In classical music this would be the conductor, orchestra, soloists. In an audio book it would be the actor who did the reading. In popular music this is typically the same as the ARTIST and is omitted.
258        Performer = "PERFORMER",
259        /// Copyright attribution, e.g., '2001 Nobody's Band' or '1999 Jack Moffitt'
260        Copyright = "COPYRIGHT",
261        /// License information, for example, 'All Rights Reserved', 'Any Use Permitted', a URL to a license such as a Creative Commons license (e.g. "creativecommons.org/licenses/by/4.0/"), or similar.
262        License = "LICENSE",
263        /// Name of the organization producing the track (i.e. the 'record label')
264        Organization = "ORGANIZATION",
265        /// A short text description of the contents
266        Description = "DESCRIPTION",
267        /// A short text indication of music genre
268        Genre = "GENRE",
269        /// Date the track was recorded
270        Date = "DATE",
271        /// Location where track was recorded
272        Location = "LOCATION",
273        /// Contact information for the creators or distributors of the track. This could be a URL, an email address, the physical address of the producing label.
274        Contact = "CONTACT",
275        /// ISRC number for the track; see [the ISRC intro page](https://isrc.ifpi.org/) for more information on ISRC numbers.
276        Isrc = "ISRC",
277    }
278}
